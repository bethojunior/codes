let elementProperty = new ElementProperty();
window.document.addEventListener("scroll" , function(){
    console.log(pageYOffset);
    if(pageYOffset > 500) {
        elementProperty.getElement('.nav-home', nav => {
            nav.style.opacity = '0.6';
        });
    }else{
        elementProperty.getElement('.nav-home', nav => {
            nav.style.opacity = '1';
        });
    }
});
